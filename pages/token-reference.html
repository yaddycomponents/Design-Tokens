import type { ThemeConfig } from 'antd';

const theme: ThemeConfig = {
  token: {
    // ── Color ─────────────────────────────────────────────
    colorPrimary:          '#531dab',
    colorSuccess:          '#52c41a',
    colorWarning:          '#ffa940',
    colorError:            '#ff4d4f',
    colorInfo:             '#1677ff',
    colorText:             '#262626',
    colorTextSecondary:    '#737373',
    colorTextTertiary:     '#bfbfbf',
    colorTextDisabled:     '#d9d9d9',
    colorBgBase:           '#ffffff',
    colorBgLayout:         '#fafafa',
    colorBorder:           '#d9d9d9',
    colorBorderSecondary:  '#f0f0f0',

    // ── Typography ────────────────────────────────────────
    fontFamily:            "'Inter', sans-serif",
    fontFamilyCode:        "'IBM Plex Mono', monospace",
    fontSize:              14,
    fontWeightStrong:      600,

    // ── Spacing ───────────────────────────────────────────
    sizeUnit:              4,
    sizeStep:              4,

    // ── Border Radius ─────────────────────────────────────
    borderRadius:          6,
    borderRadiusXS:        2,

    controlHeightLG:       40,
    controlHeightXL:       48,
  },
};

export default theme;
